      if (ww > maxW) maxW = ww;
    });
    wrap.style.width = maxW + 'px';
  }
  setTimeout(setWrapWidth, 100);
  window.addEventListener('resize', setWrapWidth);

  let cur = 0;
  const total = words.length - 1; // last is duplicate of first
  setInterval(() => {
    cur = (cur + 1) % total;
    inner.style.transform = `translateY(${-cur * 1.06}em)`;
  }, 2400);
}

// FAQ
function faq(btn) {
  const item = btn.closest('.faq-item');
  const wasOpen = item.classList.contains('open');
  document.querySelectorAll('.faq-item.open').forEach(i => i.classList.remove('open'));
  if (!wasOpen) item.classList.add('open');
}

// Form
function submit(e) {
  e.preventDefault();
  const btn = e.target.querySelector('[type=submit]');
  const orig = btn.innerHTML;
  btn.innerHTML = 'Отправлено ✓';
  btn.style.cssText = 'background:var(--pink);color:#fff;pointer-events:none';
  setTimeout(() => { btn.innerHTML = orig; btn.style.cssText = ''; e.target.reset(); }, 3000);
}
